# SellerAI sync — авто-синхронизация WB↔Ozon↔ЯМ (Amsterdam VPS)
# Установка: crontab /opt/sellerai-sync/deploy/crontab.txt
# Логи: /opt/sellerai-sync/logs/

# ВАЖНО: orders/stocks/cards пишут общий леджер -> общий lock (/tmp/sai-ledger.lock),
# чтобы исключить lost-update при пересечении тиков. flock -n = пропустить тик, если занято
# (ничего не теряется: stocks-reconcile сам обработает заказы, что пропустил order-loop).
# order-loop — заказ→сквозной синхрон остатка + Telegram-алерт, каждую минуту
* * * * * flock -n /tmp/sai-ledger.lock /usr/bin/node /opt/sellerai-sync/sync/dist/orchestrator.js orders --apply >> /opt/sellerai-sync/logs/cron-orders.log 2>&1
# full-reconcile остатков (пополнения, ручные правки, выравнивание) — каждые 30 мин
# flock -w 15: подождать ledger.lock до 15с (order-тик держит его ~1с), чтобы не скипать reconcile при коллизии в :00/:30
*/30 * * * * flock -w 15 /tmp/sai-ledger.lock /usr/bin/node /opt/sellerai-sync/sync/dist/orchestrator.js stocks --apply >> /opt/sellerai-sync/logs/cron-stocks.log 2>&1
# авто-создание недостающих карточек — каждые 2 часа
17 */2 * * * flock -w 15 /tmp/sai-ledger.lock /usr/bin/node /opt/sellerai-sync/sync/dist/orchestrator.js cards --apply >> /opt/sellerai-sync/logs/cron-cards.log 2>&1
# дрейф цен (режим A vs факт) — ОТКЛЮЧЁН 2026-07-13 по решению пользователя (хронический дрейф спамил одним списком).
# Включить обратно: раскомментировать и переустановить crontab.
# 0 9 * * * /usr/bin/node /opt/sellerai-sync/sync/dist/orchestrator.js prices >> /opt/sellerai-sync/logs/cron-prices.log 2>&1
# health-сводка — ежедневно 08:30
30 8 * * * /usr/bin/node /opt/sellerai-sync/sync/dist/orchestrator.js reconcile >> /opt/sellerai-sync/logs/cron-reconcile.log 2>&1
# приёмка себестоимости через Telegram (/cost) — каждую минуту. Отдельный lock: пишет costs.json, не леджер
* * * * * flock -n /tmp/sai-intake.lock /usr/bin/node /opt/sellerai-sync/sync/dist/orchestrator.js intake >> /opt/sellerai-sync/logs/cron-intake.log 2>&1
# месячный P&L (выручка − комиссии − COGS) — 1-го числа в 10:00
0 10 1 * * /usr/bin/node /opt/sellerai-sync/sync/dist/orchestrator.js finance >> /opt/sellerai-sync/logs/cron-finance.log 2>&1
